void quicksort(int a[]  , int l , int r){
	int i = l , j = r , flag = a[(l+r)/2] , tmp;
	do{
		while(a[i]<flag) i++;
		while(a[j]>flag) j--;
		if(i <= j){
			tmp = a[i];a[i] = a[j] ;a[j] = tmp;
			i++;j--;
		}
	}while(i<=j);
	if(l < j) quicksort(a,l,j);
	if(i < r) quicksort(a,i,r);
}

}
